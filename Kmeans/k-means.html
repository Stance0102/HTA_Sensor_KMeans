# 測試 K = 1~9，選擇迅速下降轉為平緩的點（手肘法）
kmeans_list = [KMeans(n_clusters=k).fit(data)
               for k in range(1, 10)]

print(inertias)

# 這裡就畫圖

plt.ylabel("Inertia", fontsize=14)
plt.axis([1, 9, 0, inertias[0]])
plt.show()
